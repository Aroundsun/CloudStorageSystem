cmake_minimum_required(VERSION 3.10)
project(ServerTest)

set(CMAKE_CXX_STANDARD 17)

# 生成 test 可执行文件
add_executable(test Test.cpp base64.cpp)
target_link_libraries(test pthread stdc++fs jsoncpp bundle event)

# 生成 gdb_test 可执行文件（带调试信息）
add_executable(gdb_test Test.cpp)
target_link_libraries(gdb_test pthread stdc++fs jsoncpp bundle event)
target_compile_options(gdb_test PRIVATE -g)
